function stripAddonForDisplay(data) {
  if (data == null || typeof data !== 'object' || Array.isArray(data)) return data;
  const HIDDEN = new Set(['平行演化']);
  function stripDeep(value) {
    if (value == null || typeof value !== 'object') return value;
    if (Array.isArray(value)) return value.map(stripDeep);
    const next = {};
    for (const [k, v] of Object.entries(value)) {
      if (HIDDEN.has(k)) continue;
      next[k] = stripDeep(v);
    }
    return next;
  }
  return stripDeep(_.cloneDeep(data));
}

const addonData = getvar('addon_data');
const addonUi = getvar('addon_ui') || {};
const planeMerge = addonUi.位面交汇 === true;
const replicaKey = String(`{{replica:val}}`).trim();
const isEmpty = !addonData || Object.keys(addonData || {}).length === 0;

let displayData = null;
let statusNote = '';

if (isEmpty) {
  statusNote = '[addon 变量状态] 当前 addon_data 为空，请先在 addon 控制台新建世界；AI 不会创建世界键。';
} else if (planeMerge) {
  displayData = stripAddonForDisplay(addonData);

} else {
  const stripped = stripAddonForDisplay(addonData);

    statusNote = `[addon 变量状态] 未找到世界「${replicaKey}」。`;
    displayData = {};
  } else {
    displayData = { [replicaKey]: world };
  }
}
